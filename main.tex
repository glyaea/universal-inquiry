\begin{metadefinition}[Formula]
	If $a,b$ are variables,
	then $a\in b$ is a formula.
	If $p$ is a formula,
	then $\neg p$ is a formula.
	If $p,q$ are formulas,
	then $p\land q$ is a formula.
	If $x$ is a variable,
	and $p$ is a formula,
	then $\exists x(p)$ is a formula.
\end{metadefinition}

\begin{metadefinition}
	If $p(x)$ is a formula,
	and $y$ is a variable,
	then $p(y)$ is $p(x)$ with each occurrence of $x$ replaced with $y$.
\end{metadefinition}

\begin{notation}[Disjunction]
$$
	p\lor q
	\equiv
	\neg(\neg p\land\neg q)
$$
\end{notation}

\begin{notation}[Implication]
$$
	p\Rightarrow q
	\equiv
	\neg(p\land\neg q)
$$
\end{notation}

\begin{notation}[Equivalence]
$$
	p\Leftrightarrow q
	\equiv
	(p\Rightarrow q)
	\land
	(q\Rightarrow p)
$$
\end{notation}

\begin{notation}[Exclusive Disjunction]
$$
	p\veebar q
	\equiv
	\neg(p\Leftrightarrow q)
$$
\end{notation}

\begin{notation}[Negated Existential Quantification]
$$
	\nexists
	\equiv
	\neg\exists
$$
\end{notation}

\begin{notation}[Bounded Existential Quantification]
$$
	\exists x\in A:(p)
	\equiv
	\exists x:(x\in A\land p)
$$
\end{notation}

\begin{notation}[Universal Quantification]
$$
	\forall x:(p)
	\equiv
	\nexists x:(\neg p)
$$
\end{notation}

\begin{notation}[Negated Universal Quantification]
$$
	\nforall
	\equiv
	\neg\forall
$$
\end{notation}

\begin{notation}[Bounded Universal Quantification]
$$
	\forall x\in A:(p)
	\equiv
	\forall x:(x\in A\Rightarrow p)
$$
\end{notation}

\begin{notation}[Equality]
$$
	X=Y
	\equiv
	\forall a:(
		a\in X
		\Leftrightarrow
		a\in Y
	)
	\land
	\forall B:(
		X\in B
		\Leftrightarrow
		Y\in B
	)
$$
\end{notation}

\begin{notation}[Negated Equality]
$$
	a\neq b
	\equiv
	\neg(a=b)
$$
\end{notation}

\begin{notation}[Unique Quantification]
$$
	\exists!x:(
		p(x)
	)
	\equiv
	\exists x:(
		p(x)
		\land
		\forall y:(
			p(y)
			\Rightarrow
			y=x
		)
	)
$$
\end{notation}

\begin{axiom}[Empty Set]
$$
	\exists n,
	\nexists x:(
		x\in n
	)
$$
\end{axiom}

\begin{definition}[Empty Set]
$$
	n=\varnothing
	\Leftrightarrow
	\nexists x:(
		x\in n
	)
$$
\end{definition}

\begin{axiom}[Extensionality]
$$
	\forall A,
	\forall B:(
		\forall x:(
			x\in A
			\Leftrightarrow
			x\in B
		)
		\Rightarrow
		A=B
	)
$$
\end{axiom}

\begin{axiom}[Pairing]
$$
	\forall a,
	\forall b,
	\exists C,
	\forall c:(
		c\in C
		\Leftrightarrow
		(
			c=a
			\lor
			c=b
		)
	)
$$
\end{axiom}

\begin{definition}[Couple]
$$
	C=\{a,b\}
	\Leftrightarrow
	\forall c:(
		c\in C
		\Leftrightarrow
		(
			c=a
			\lor
			c=b
		)
	)
$$
\end{definition}

\begin{notation}[Singleton]
$$
	\{a\}
	\equiv
	\{a,a\}
$$
\end{notation}

\begin{axiom}[Union]
$$
	\forall B,
	\exists U,
	\forall c:(
		c\in U
		\Leftrightarrow
		\exists P\in B:(
			c\in P
		)
	)
$$
\end{axiom}

\begin{definition}[Union]
$$
	U=\bigcup B
	\Leftrightarrow
	\forall c:(
		c\in U
		\Leftrightarrow
		\exists P\in B:(
			c\in P
		)
	)
$$
\end{definition}

\begin{definition}[Binary Union]
$$
	A\cup B
	=
	\bigcup\{A,B\}
$$
\end{definition}

\begin{definition}[Intersection]
$$
	I=\bigcap B
	\Leftrightarrow
	B\neq\varnothing
	\land
	\forall c:(
		c\in I
		\Leftrightarrow
		\forall P\in B:(
			c\in P
		)
	)
$$
\end{definition}

\begin{definition}[Binary Intersection]
$$
	A\cap B
	=
	\bigcap\{A,B\}
$$
\end{definition}

\begin{axiom}[Power Set]
$$
	\forall X,
	\exists P,
	\forall S:(
		S\in P
		\Leftrightarrow
		\forall s\in S:(
			s\in X
		)
	)
$$
\end{axiom}

\begin{definition}[Power Set]
$$
	P=\mathcal{P}(X)
	\Leftrightarrow
	\forall S:(
		S\in P
		\Leftrightarrow
		\forall s\in S:(
			s\in X
		)
	)
$$
\end{definition}

\begin{definition}[Subset]
$$
	S\subseteq X
	\Leftrightarrow
	S\in\mathcal{P}(X)
$$
\end{definition}

\begin{definition}[Successor]
$$
	\operatorname{successor}(x)
	=
	x\cup\{x\}
$$
\end{definition}

\begin{definition}[Inductive Set]
$$
	\operatorname{Inductive}(I)
	\Leftrightarrow
	\varnothing\in I
	\land
	\forall i\in I:(
		\operatorname{successor}(i)\in I
	)
$$
\end{definition}

\begin{axiom}[Infinity]
$$
	\exists I:(
		\operatorname{Inductive}(I)
	)
$$
\end{axiom}

\begin{definition}[Pair]
$$
	(a,b)
	=
	\{\{a\},\{a,b\}\}
$$
\end{definition}

\begin{axiom}[Foundation]
$$
	\forall X:(
		X\neq\varnothing
		\Rightarrow
		\exists x\in X:(
			x\cap X
			=
			\varnothing
		)
	)
$$
\end{axiom}

\begin{definition}[Set of Free Variables]
\begin{align*}
	\operatorname{free}(a\in b)
	&=
	\{a,b\}
	\\
	\operatorname{free}(\neg p)
	&=
	\operatorname{free}(p)
	\\
	\operatorname{free}(p\land q)
	&=
	\operatorname{free}(p)
	\cup
	\operatorname{free}(q)
	\\
	\operatorname{free}(
		\exists x:(p)
	)
	&=
	\operatorname{free}(p)
	\setminus
	\{x\}
\end{align*}
\end{definition}

\begin{axiom}[Replacement]
Let $B\notin\operatorname{free}(p)$.
$$
	\forall A:(
		\forall x\in A,\,
		\exists!y:(
			p
		)
		\Rightarrow
		\exists B,
		\forall y:(
			y\in B
			\Leftrightarrow
			\exists x\in A:(p)
		)
	)
$$
\end{axiom}

\begin{definition}[Set Builder]
Let $X\notin\operatorname{free}(p)$.
$$
	X=\{a\in A\mid p\}
	\Leftrightarrow
	\forall x:(
		x\in X
		\Leftrightarrow
		(
			x\in A
			\land
			p
		)
	)
$$
\end{definition}

\begin{definition}[Cartesian Product]
$$
	A\times B
	=
	\{
		(a,b)
		\mid
		a\in A
		\land
		b\in B
	\}
$$
\end{definition}

\begin{definition}[Set of Functions]
$$
	Y^{X}
	=
	\{
		f\subseteq X\times Y
		\mid
		\forall x\in X,
		\exists!y\in Y:(
			(x,y)\in f
		)
	\}
$$
\end{definition}

\begin{definition}[Function]
\begin{align*}
	f:X\to Y
	&\Leftrightarrow
	f\in Y^{X}
	\\
	f(x)=y
	&\Leftrightarrow
	(x,y)\in f
\end{align*}
\end{definition}

\begin{definition}[Indexed Product]
$$
	\prod B
	=
	\mleft\{
		c:B\to\bigcup B
		\setmiddle
		\forall P\in B:(
			c(P)\in P
		)
	\mright\}
$$
\end{definition}

\begin{axiom}[Choice]
$$
	\forall O:\mleft(
		\forall o\in O:(
			o\neq\varnothing
		)
		\Rightarrow
		\prod O\neq\varnothing
	\mright)
$$
\end{axiom}

\begin{definition}[Exclusion]
$$
	A\setminus B
	=
	\{
		x\in A
		\mid
		x\notin B
	\}
$$
\end{definition}

\begin{definition}[Set of Natural Numbers]
$$
	N=\N
	\Leftrightarrow
	\forall n:(
		n\in N
		\Leftrightarrow
		\forall I:(
			\operatorname{Inductive}(I)
			\Rightarrow
			n\in I
		)
	)
$$
\end{definition}

\begin{definition}[Strict Order of Natural Numbers]
Let $a,b\in\N$.
$$
	a<b
	\Leftrightarrow
	a\in b
$$
\end{definition}

\begin{notation}[Non-Strict Order]
$$
	a\leq b
	\equiv
	a<b
	\lor
	a=b
$$
\end{notation}

\begin{align*}
	0

\begin{definition}[Tuple]
Let $n\in\N\setminus\{0\}$.
$$
	(x_{1},\dots,x_{n})
	=
	\begin{cases}
	x_{1}, & \text{if}\ n=1 \\
	(x_{1},x_{2}), & \text{if}\ n=2 \\
	((x_{1},\dots,x_{n-1}),x_{n}) & \text{otherwise}
	\end{cases}
$$
\end{definition}
